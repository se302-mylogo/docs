20 def assertAccessToken(self, token: str, data: Mapping, ignores: List[str] = None): 

21 if ignores is None: 

22 ignores = ['password', 'avatar'] 

23 access_token = AccessToken(token) 

24 keys = [k for k in data.keys() if k not in ignores] 

25 payload_keys = [self.to_camel_case(k) for k in keys] 

26 self.assertEqual(self.filter_keys(access_token.payload, payload_keys, camel=True), 

27 self.filter_keys(data, keys, camel=True))
